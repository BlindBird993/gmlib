set(BLAS_HEADER_FILE "${CMAKE_CURRENT_LIST_DIR}/gmblasdmatrixincludes.h")
file(WRITE ${BLAS_HEADER_FILE} "// This file was generated by CMake.\n")

set(BLAS_ROOT_DIR ${CMAKE_CURRENT_LIST_DIR})

macro(addBlasHeaderLines)
  foreach(SOURCE ${ARGV})
    file(APPEND ${BLAS_HEADER_FILE} "\n${SOURCE}")
  endforeach(SOURCE ${ARGV})
endmacro(addBlasHeaderLines)

macro(addBlasHeaders)
  foreach(SOURCE ${ARGV})
    file(RELATIVE_PATH REL_PATH ${BLAS_ROOT_DIR} "${CMAKE_CURRENT_LIST_DIR}/${SOURCE}")
	file(APPEND ${BLAS_HEADER_FILE} "\n#include \"${REL_PATH}\"")
  endforeach(SOURCE ${ARGV})
endmacro(addBlasHeaders)
